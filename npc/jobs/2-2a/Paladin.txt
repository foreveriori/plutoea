valkyrie,53,39,3	script	聖殿十字軍#Valkyrie	752,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[聖殿十字軍]";
			mes "恭喜你轉生了!";
			mes "這是戰士們的光榮!";
			close;
		}
		mes "[聖殿十字軍]";
		mes "別忘了聖戰時刻將至，";
		mes "請不要鬆懈鍛鍊!";
		mes "願奧丁神保佑你!";
		mes "請立刻行動!";
		close;
	}
	else if (ADVJOB == Job_Paladin && Class == Job_Swordman_High && JobLevel > 39) {
		mes "[聖殿十字軍]";
		mes "恢復本來面貌、";
		mes "給世界帶來新的曙光的時候來臨了";
		mes "請以嶄新的姿態繼續你的旅程吧";
		next;
		mes "[聖殿十字軍]";
		mes "要轉職嗎?";
		next;
		if (select("不要:轉職成為聖殿十字軍") == 1) {
			mes "[聖殿十字軍]";
			mes "準備好了的話，";
			mes "請隨時找我吧";
			mes "祝你好運!";
			close;
		}
		if (SkillPoint) {
			mes "[聖殿十字軍]";
			mes "好像學習的還不夠多，";
			mes "學會所有技能之後再來吧";
			close;
		}
		jobchange Job_Paladin;
		set ADVJOB,0;
		mes "[聖殿十字軍]";
		mes "恭喜你轉職成功";
		mes "請回到陸地上，";
		mes "用你的保護弱者吧";
		close;
	}
	mes "[聖殿十字軍]";
	mes "歡迎來到榮譽的巴哈拉，";
	mes "好好的休息再回去吧，";
	mes "戲給戰士們光榮!";
	close;
}
