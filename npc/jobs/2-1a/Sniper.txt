valkyrie,44,55,5	script	神射手#Valkyrie	727,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[神射手]";
			mes "恭喜你轉生了!";
			mes "這是戰士們的光榮!";
			close;
		}
		mes "[神射手]";
		mes "^6B8C21來，轉職考試要開始了，";
		mes "在無數的魔物當中，要抓出名字是";
		mes "'轉職考試用魔物'的傢伙，";
		mes "***** 抓的時候請注意陷阱 *****^000000";
		mes ".......";
		close;
	}
	else if (ADVJOB == Job_Sniper && Class == Job_Archer_High && JobLevel > 39) {
		mes "[神射手]";
		mes "恢復本來面貌、";
		mes "給世界帶來新的曙光的時候來臨了";
		mes "請以嶄新的姿態繼續你的旅程吧";
		next;
		mes "[神射手]";
		mes "要轉職嗎?";
		next;
		if (select("不要:轉職成為神射手") == 1) {
			mes "[神射手]";
			mes "準備好了的話，";
			mes "請隨時找我吧";
			mes "祝你好運!";
			close;
		}
		if (SkillPoint) {
			mes "[神射手]";
			mes "好像學習的還不夠多，";
			mes "學會所有技能之後再來吧";
			close;
		}
		jobchange Job_Sniper;
		set ADVJOB,0;
		mes "[神射手]";
		mes "恭喜你轉職成功";
		mes "請回到陸地上，";
		mes "希望在你視線中的獵物都逃不了";
		close;
	}
	mes "[神射手]";
	mes "歡迎來到榮譽的巴哈拉，";
	mes "好好的休息再回去吧，";
	mes "戲給戰士們光榮!";
	close;
}
