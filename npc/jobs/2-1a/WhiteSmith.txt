valkyrie,44,50,5	script	神工匠#Valkyrie	731,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[神工匠]";
			mes "恭喜你轉生成功";
			mes "戲給戰士們光榮!!";
			close;
		}
		mes "[神工匠]";
		mes "這是收據，我想你也辛苦了";
		mes ""+strcharinfo(0)+"...轉送，謝謝你囉";
		mes "......";
		close;
	}
	else if (ADVJOB == Job_WhiteSmith && Class == Job_Merchant_High && JobLevel > 39) {
		mes "[神工匠]";
		mes "恢復本來面貌、";
		mes "給世界帶來新的曙光的時候來臨了";
		mes "請以嶄新的姿態繼續你的旅程吧";
		next;
		mes "[神工匠]";
		mes "要轉職嗎?";
		next;
		if (select("不要:是的") == 1) {
			mes "[神工匠]";
			mes "準備好了的話，";
			mes "請隨時找我吧";
			mes "祝你好運!";
			close;
		}
		if (SkillPoint) {
			mes "[神工匠]";
			mes "請把技能點全部加完再來吧";
			close;
		}
		jobchange Job_Whitesmith;
		set ADVJOB,0;
		mes "[神工匠]";
		mes "恭喜你轉職成功";
		mes "請回到陸地上，";
		mes "用你的技術做出更棒的武器吧";
		close;
	}
	mes "[神工匠]";
	mes "歡迎來到榮譽的巴哈啦，";
	mes "好好的休息再回去吧，";
	mes "戲給戰士們光榮!!";
	next;
	mes "[神工匠]";
	mes "歡迎凱旋歸來";
	mes "在這裡好好休息一下吧";
	mes "戰士們的榮耀!";
	close;
}
