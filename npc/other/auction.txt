function	script	F_AuctionWarper	{
	mes "[拍賣場 服務人員]";
	mes "請問你要去拍賣場嗎?";
	next;
	if (select("是的:不是") == 1) {
		mes "[拍賣場 服務人員]";
		mes "那麼祝你一路順風~";
		close2;
		switch(getarg(0)) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[拍賣場 服務人員]";
	mes "那麼下一次請你再來呦";
	close;
}
