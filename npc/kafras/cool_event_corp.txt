hugel.gat,88,168,4	script	准它 企劃社 服務人員::CoolEventCorpStaffHugel	874,{

	mes "[准它 企劃社 服務人員]";
	mes "歡迎光臨，准它";
	mes "企畫設永遠秉持著最好服務的，";
	mes "請問您需要哪一種服務呢?";
	next;

	switch(select("儲存服務:倉庫服務:手推車服務:結束")){
	case 1:
		mes "[准它 企劃社 服務人員]";
		mes "已經儲存好了，";
		mes "謝謝您的使用";
		emotion e_thx;
		savepoint "hugel",96,155;
		close;
	case 2:
		mes "[准它 企劃社 服務人員]";
		if(basicskillcheck() && getskilllv(1) < 6){
			mes "I'm sorry, but you";
			mes "need the Novice's";
			mes "Basic Skill Level 6 to";
			mes "use the Storage Service.";
			emotion e_sry;
			close;
		}
		if(Zeny < 40){
			mes "顧客，您支付的金額不足啊，";
			mes "倉庫使用費是 40金幣";
			emotion e_cash;
			close;
		}
		set Zeny, Zeny - 40;
		set RESRVPTS, RESRVPTS + 4;
		mes "倉庫將打開了，";
		mes "感謝您的使用";
		close2;
		openstorage;
		end;
	case 3:
		mes "[准它 企劃社 服務人員]";
		if(BaseClass != Job_Merchant){
			mes "很抱歉，";
			mes "手推車是商人和鐵匠、鍊金術師專用的服務";
			emotion e_sry;
			close;
		}
		if(getskilllv(39) == 0){
			mes "You can only rent a cart after";
			mes "learning the Pushcart Skill.";
			close;
		}
		if(checkcart() == 1){
			mes "You already have";
			mes "a Pushcart equipped.";
			mes "Unfortunately, we can't";
			mes "rent more than one to";
			mes "each customer at a time.";
			close;
		}
		if(countitem(7061)){
			delitem 7061,1;
			setcart;
			close;
		}
		mes "The Pushcart rental";
		mes "fee is 800 Zeny. Would";
		mes "you like to rent a Pushcart?";
		next;
		if(select("Rent a Pushcart:Cancel") == 1){
			if(Zeny < 800){
				mes "[准它 企劃社 服務人員]";
				mes "I'm sorry, but you";
				mes "don't have enough";
				mes "Zeny to pay the Pushcart";
				mes "rental fee of 800 Zeny.";
				emotion e_cash;
				close;
			} else {
				set Zeny, Zeny - 800;
				set RESRVPTS, RESRVPTS + 80;
				setcart;
				close;
			}
		}
		close;
	case 4:
	default:
		mes "[准它 企劃社 服務人員]";
		mes "我們准它 企劃社";
		mes "永遠提供最好服務的，若有不便之處的";
		mes "話請隨時告知我們，";
		mes "下一次預計以更好的服務與你見面";
		close;
	}
}
