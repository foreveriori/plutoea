	set @wrpC$[1], @wrpD$[1]+"　-> "+@wrpP[1]+" z";
	set @wrpC$[2], @wrpD$[2]+"　-> "+@wrpP[2]+" z";
	set @wrpC$[3], @wrpD$[3]+"　-> "+@wrpP[3]+" z";
	set @wrpC$[4], @wrpD$[4]+"　-> "+@wrpP[4]+" z";
	set @wrpC$[5], @wrpD$[5]+"　-> "+@wrpP[5]+" z";
	set @wrpC$[6], "取消";
	setarray @viewpX[0], 151, 29, 282, 152;
	setarray @viewpY[0], 29, 207, 200, 326;
	return;
}
