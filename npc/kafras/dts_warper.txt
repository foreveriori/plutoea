// 商權投票
//-------------------------------------------
// 規則說明：
// 1.開始投票後每２４個小時會檢查一次票數
//   當雙方的得票都超過２０票後，則由得票數
//   高的一方獲勝；反之任一方未達２０票則
//   重新計算２４個小時直到雙方都超過２０票為止
// 2.剛開始雙方都只有一個服務點可供傳送，但是
//   只要任一方在這次投票中敗選，下次投票時
//   就會有新的傳送點出現；反之此次勝選，下次
//   就會減少一個服務點，不過最少保留一個
// 3.每六天重新投票一次。所以這腳本適合常設型的伺服器
//-------------------------------------------
-	script	VoteEvent	-1,{
OnInit:
	initnpctimer;
	end;
OnTimer60000:
	set $VoteTime,$VoteTime+1;
	if($VoteResult == 0) {
		if($VoteTime >= 1440) {
			if($VoteKafra < 20 && $VoteZonda < 20)
				set $VoteTime,0;
			else {
				if($VoteKafra >= $VoteZonda) {
					if($VoteZondaP < 2)
						set $VoteZondaP,$VoteZondaP+1;
					set $VoteResult,1;
					set $VoteResult$,"卡普拉服務";
				}
				else {
					if($VoteKafraP < 2)
						set $VoteKafraP,$VoteKafraP+1;
					set $VoteResult,2;
					set $VoteResult$,"准它企劃社";
				}
				announce "商權投票已經結束了，請靜待結果發表",8;
				sleep 5000;
				announce "發表商權投票的投票結果..",8;
				sleep 5000;
				announce "得票數最多的是'" +$VoteResult$+ "'，恭喜'" +$VoteResult$+ "'取得了營運權",8;
				sleep 5000;
				announce "" +$VoteResult$+ " : 今後也請多多指教，感謝大家的支持",8;
				set $VoteTime,0;
				initnpctimer;
				end;
			}
		}
		if($VoteTime % 120 == 0)
			announce "目前舉行商權投票中，請多多參與",8;
	}
	else if($VoteTime >= 8640) {
		if($VoteResult == 1 && $VoteKafraP > 0)
			set $VoteKafraP,$VoteKafraP-1;
		if($VoteResult == 2 && $VoteZondaP > 0)
			set $VoteZondaP,$VoteZondaP-1;
		set $VoteTime,0;
		set $VoteResult,0;
		set $VoteKafra,0;
		set $VoteZonda,0;
		set $VoteResult$,"";
		set $VoteNumber,$VoteNumber+1;
		announce "目前舉行商權投票中，請多多參與",8;
	}
	initnpctimer;
	end;
}
